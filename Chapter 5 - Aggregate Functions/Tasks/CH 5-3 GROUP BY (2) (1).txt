1 DISPLAY DEPARTMENT_ID WISE NO. OF EMPLOYEES
SELECT COUNT(*) FROM EMPLOYEES GROUP BY DEPARTMENT_ID;

2 DISPLAY JOB_WISE LOWEST AND HIGHEST SALARY
SELECT MIN(SALARY) , MAX(SALARY) FROM EMPLOYEES GROUP BY JOB_ID;

3 DISPLAY MANAGER_ID WISE AVERAGE SALARY
SELECT AVG(SALARY) FROM EMPLOYEES GROUP BY MANAGER_ID;

4 DISPLAY DEPARTMENT_ID AND JOB_ID WISE NO. OF EMPLOYEES
SELECT COUNT(*) FROM EMPLOYEES GROUP BY DEPARTMENT_ID , JOB_ID;

5 DISPLAY HIRE YEAR WISE NO. OF EMPLOYEES
SELECT COUNT(*) FROM EMPLOYEES GROUP BY SUBSTR(HIRE_DATE,-2);

6 DISPLAY QUARTER WISE TOTAL OF SALARY
SELECT SUM(SALARY) FROM EMPLOYEES GROUP BY TO_CHAR(HIRE_DATE,'Q');

7 DISPLAY DEPARTMENT_ID WISE, JOB_ID WISE LOWEST AND HIGHEST SALARY
SELECT MIN(SALARY) , MAX(SALARY) FROM EMPLOYEES GROUP BY DEPARTMENT_ID , JOB_ID;

8 DISPLAY MANAGER_ID WISE , JOB_ID WISE AVERAGE SALARY AND TOTAL OF SALARY
SELECT AVG(SALARY) , SUM(SALARY) FROM EMPLOYEES GROUP BY MANAGER_ID , JOB_ID;

9 DISPLAY DEPARTMENT_ID WISE , MANAGER_ID WISE COUNT OF EMPLOYEES
SELECT COUNT(*) FROM EMPLOYEES GROUP BY DEPARTMENT_ID , MANAGER_ID;

10 DISPLAY DEPARTMENT_ID WISE , JOB_ID WISE , MANAGER_ID WISE LOWEST SALARY, AVERAGE SALARY AND HIGHEST SALARY
SELECT DEPARTMENT_ID , JOB_ID , MANAGER_ID  , MIN(SALARY) , AVG(SALARY) , MAX(SALARY) FROM EMPLOYEES GROUP BY DEPARTMENT_ID , MANAGER_ID;

11 DISPLAY ONLY DATE WISE (IGNORE MONTH AND YEAR) HIGHEST AND LOWEST SALARY
SELECT MAX(SALARY) , MIN(SALARY) FROM EMPLOYEES GROUP BY SUBSTR(HIRE_DATE,1,2);

12 DISPLAY DAY WISE (MON,TUE...) AVERAGE SALARY
SELECT AVG(SALARY) FROM EMPLOYEES GROUP BY TO_CHAR(HIRE_DATE,DAY);

13 DISPLAY MONTH WISE (IGNORE DATE AND YEAR) TOTAL SALARY
SELECT SUM(SALARY) FROM EMPLOYEES GROUP BY TO_CHAR(HIRE_DATE,MM);

14 DISPLAY MONTH WISE (IGNORE DATE AND YEAR) TOTAL SALARY, SORT ROWS ON MONTH NUBMER.
SELECT TO_CHAR(HIRE_DATE,MM) , SUM(SALARY) FROM EMPLOYEES GROUP BY TO_CHAR(HIRE_DATE,MM) ORDER BY 1;

15 DISPLAY DAY WISE (MON,TUE...) AVERAGE SALARY, SORT ROWS ON DAY NUMBER OF WEEK.
SELECT AVG(SALARY) FROM EMPLOYEES GROUP BY TO_CHAR(HIRE_DATE,'DAY') ORDER BY TO_CHAR(HIRE_DATE,'D');

16 DISPLAY ALPHABET RANGE WISE COUNT IN FIRST_NAME SPELLING. (A-D , E-G , I-N, O-T, U-Z)
SELECT COUNT(CASE WHEN SUBSTR(FIRST_NAME,1,1) BETWEEN 'A' AND 'D' THEN '*' END) A_D ,
       COUNT(CASE WHEN SUBSTR(FIRST_NAME,1,1) BETWEEN 'E' AND 'G' THEN '*' END) E_G ,
	   COUNT(CASE WHEN SUBSTR(FIRST_NAME,1,1) BETWEEN 'I' AND 'N' THEN '*' END) I_N ,
	   COUNT(CASE WHEN SUBSTR(FIRST_NAME,1,1) BETWEEN 'O' AND 'T' THEN '*' END) O_T ,
	   COUNT(CASE WHEN SUBSTR(FIRST_NAME,1,1) BETWEEN 'U' AND 'Z' THEN '*' END) U_Z 
FROM EMPLOYEES;
