1 DISPLAY EMPLOYEE'S FIRST_NAME AND MANAGER'S FIRST_NAME
SELECT E.FIRST_NAME EMP_FIRST_NAME , M.FIRST_NAME MGR_FIRST_NAME
FROM EMPLOYEES E
JOIN EMPLOYEES M ON M.EMPLOYEE_ID = E.MANAGER_ID;

2 DISPLAY EMPLOYEE'S FIRST_NAME AND LAST_NAME  WITH MANAGER'S FIRST_NAME AND LAST_NAME
SELECT E.FIRST_NAME EMP_FIRST_NAME , E.LAST_NAME EMP_LAST_NAME , M.FIRST_NAME MGR_FIRST_NAME , M.LAST_NAME MGR_LAST_NAME
FROM EMPLOYEES E 
JOIN EMPLOYEES M ON M.EMPLOYEE_ID = E.MANAGER_ID;

3 DISPLAY EMPLOYEE'S FIRST_NAME , SALARY WITH MANAGER'S FIRST_NAME AND SALARY
SELECT E.FIRST_NAME EMP_FIRST_NAME , E.SALARY EMP_SALARY , M.FIRST_NAME MGR_FIRST_NAME , M.SALARY MGR_SALARY
FROM EMPLOYEES E
JOIN EMPLOYEES M ON M.EMPLOYEE_ID = E.MANAGER_ID;

4 DISPLAY EMPLOYEE'S LAST_NAME AND HIRE_DATE WITH MANAGER'S FIRST_NAME AND HIRE_DATE
SELECT E.LAST_NAME EMP_LAST_NAME , E.HIRE_DATE EMP_HIRE_DATE , M.FIRST_NAME MGR_FIRST_NAME , M.HIRE_DATE MGR_HIRE_DATE
FROM EMPLOYEES E
JOIN EMPLOYEES M ON M.EMPLOYEE_ID = E.MANAGER_ID;

5 DISPLAY EMPLOYEE'S FIRST_NAME AND JOB_TITLE WITH MANAGER' FIRST_NAME AND JOB_TITLE
SELECT E.FIRST_NAME EMP_FIRST_NAME , EJ.JOB_TITLE EMP_JOB_TITLE, M.FIRST_NAME MGR_FIRST_NAME , MJ.JOB_TITLE MGR_JOB_TITLE
FROM EMPLOYEES E 
JOIN EMPLOYEES M ON M.EMPLOYEE_ID = E.MANAGER_ID
JOIN JOBS EJ ON EJ.JOB_ID = E.JOB_ID
JOIN JOBS MJ ON MJ.JOB_ID = M.JOB_ID;

6 DISPLAY EMPLOYEE'S LAST_NAME , SALARY , GRADE_LEVEL WITH MANAGER'S FIRST_NAME , SALARY , GRADE_LEVEL
SELECT E.LAST_NAME EMP_LAST_NAME , E.SALARY EMP_SALARY , EJG.GRADE_LEVEL EMP_GRADE_LEVEL , M.FIRST_NAME MGR_FIRST_NAME , M.SALARY MGR_SALARY , MJG.GRADE_LEVEL MGR_GRADE_LEVEL
FROM EMPLOYEES E
JOIN EMPLOYEES M ON M.EMPLOYEE_ID = E.MANAGER_ID
JOIN JOB_GRADES EJG ON E.SALARY BETWEEN EJG.LOWEST_SAL AND EJG.HIGHEST_SAL
JOIN JOB_GRADES MJG ON M.SALARY BETWEEN MJG.LOWEST_SAL AND MJG.HIGHEST_SAL;

7 DISPLAY EMPLOYEE'S NAME AND EMPLOYEE'S DEPARTMENT_NAME  , MANAGER'S NAME AND MANAGER'S DEPARTMENT_NAME
SELECT E.FIRST_NAME || ' ' || E.LAST_NAME EMP_NAME , ED.DEPARTMENT_NAME EMP_DEPT_NAME , M.FIRST_NAME || ' ' || M.LAST_NAME MGR_NAME , MD.DEPARTMENT_NAME MGR_DEPT_NAME  
FROM EMPLOYEES E
JOIN EMPLOYEES M ON M.EMPLOYEE_ID = E.MANAGER_ID
JOIN DEPARTMENTS ED ON ED.DEPARTMENT_ID = E.DEPARTMENT_ID
JOIN DEPARTMENTS MD ON MD.DEPARTMENT_ID = M.DEPARTMENT_ID;

8 DISPLAY EMPLOYEE'S NAME , EMPLOYEE'S CITY , MANAGER'S NAME , MANAGER'S CITY
SELECT E.FIRST_NAME || ' ' || E.LAST_NAME EMP_NAME , EL.CITY EMP_CITY , M.FIRST_NAME || ' ' || M.LAST_NAME MGR_NAME , ML.CITY MGR_CITY  
FROM EMPLOYEES E
JOIN EMPLOYEES M ON M.EMPLOYEE_ID = E.MANAGER_ID
JOIN DEPARTMENTS ED ON ED.DEPARTMENT_ID = E.DEPARTMENT_ID
JOIN DEPARTMENTS MD ON MD.DEPARTMENT_ID = M.DEPARTMENT_ID
JOIN LOCATIONS   EL ON EL.LOCATION_ID = ED.LOCATION_ID
JOIN LOCATIONS   ML ON ML.LOCATION_ID = MD.LOCATION_ID;

9 DISPLAY EMPLOEYE'S NAME , EMPLOYEE'S REGION_NAME , MANAGER'S NAME , MANAGER'S REGION_NAME
SELECT E.FIRST_NAME || ' ' || E.LAST_NAME EMP_NAME , ER.REGION_NAME EMP_REGION_NAME , M.FIRST_NAME || ' ' || M.LAST_NAME MGR_NAME , MR.REGION_NAME MGR_REGION_NAME  
FROM EMPLOYEES E
JOIN EMPLOYEES M ON M.EMPLOYEE_ID = E.MANAGER_ID
JOIN DEPARTMENTS ED ON ED.DEPARTMENT_ID = E.DEPARTMENT_ID
JOIN DEPARTMENTS MD ON MD.DEPARTMENT_ID = M.DEPARTMENT_ID
JOIN LOCATIONS   EL ON EL.LOCATION_ID = ED.LOCATION_ID
JOIN LOCATIONS   ML ON ML.LOCATION_ID = MD.LOCATION_ID
JOIN COUNTRIES   EC ON EC.COUNTRY_ID = EL.COUNTRY_ID
JOIN COUNTRIES   MC ON MC.COUNTRY_ID = ML.COUNTRY_ID
JOIN REGIONS     ER ON EC.REGION_ID = ER.REGION_ID
JOIN REGIONS     MR ON MR.REGION_ID = MC.REGION_ID;

10 DISPLAY EMPLOYEE NAME , EMPLOYEE SALARY , EMPLOYEE GRADE_LEVEL , EMPLOYEE JOB_TITLE , MANAGER NAME , MANAGER SALARY , MANAGER GRADE_LEVEL  MANAGER JOB_TITLE
SELECT E.FIRST_NAME || ' ' || E.LAST_NAME EMP_NAME , E.SALARY EMP_SALARY , EJG.GRADE_LEVEL EMP_GRADE_LEVEL , EJ.JOB_TITLE EMP_JOB_TITLE , M.FIRST_NAME || ' ' || M.LAST_NAME MGR_NAME, M.SALARY MGR_SALARY , MJG.GRADE_LEVEL MGR_GRADE_LEVEL , EJ.JOB_TITLE EMP_JOB_TITLE 
FROM EMPLOYEES E
JOIN EMPLOYEES M ON M.EMPLOYEE_ID = E.MANAGER_ID
JOIN JOBS EJ        ON EJ.JOB_ID = E.JOB_ID
JOIN JOBS MJ        ON MJ.JOB_ID = M.JOB_ID
JOIN JOB_GRADES EJG ON E.SALARY BETWEEN EJG.LOWEST_SAL AND EJG.HIGHEST_SAL
JOIN JOB_GRADES MJG ON M.SALARY BETWEEN MJG.LOWEST_SAL AND MJG.HIGHEST_SAL;


11 DISPLAY EMPLOYEE NAME AND HIS/HER DEPARTMENTAL COLLEGUE NAME. (EVERY VALID COMBINATION SHOULD BE DISPLAYED ONCE ONLY)
SELECT E.FIRST_NAME , C.FIRST_NAME
FROM EMPLOYEES E
JOIN EMPLOYEES C ON C.DEPARTMENT_ID = E.DEPARTMENT_ID AND E.EMPLOYEE_ID > C.EMPLOYEE_ID;

12 DISPLAY THOSE WHO GETS HIGHER COMMISSION AMOUNT THAN Jonathon.
SELECT * FROM EMPLOYEES E
JOIN EMPLOYEES T ON T.COMMISSION_PCT <= E.COMMISSION_PCT
WHERE T.FIRST_NAME = 'Jonathon';
 
13 DISPLAY THOSE WHO GETS LOWER SALARY THAN Hunold.
SELECT E.FIRST_NAME , E.SALARY 
FROM EMPLOYEES E
JOIN EMPLOYEES T ON E.SALARY < T.SALARY
WHERE T.LAST_NAME = 'Hunold';

14 DISPLAY THOSE WHO WERE HIRED IN THE SAME YEAR IN WHICH Matos HIRED.
SELECT E.FIRST_NAME , E.HIRE_DATE 
FROM EMPLOYEES E
JOIN EMPLOYEES T ON SUBSTR(E.HIRE_DATE,-2) = SUBSTR(T.HIRE_DATE,-2)
WHERE T.LAST_NAME = 'Matos';

15 DISPLAY THOSE WHO REPORTS TO SAME MANAGER TO WHICH Matos REPORTS.
SELECT E.FIRST_NAME , E.MANAGER_ID
FROM EMPLOYEES E
JOIN EMPLOYEES T ON T.MANAGER_ID = E.MANAGER_ID
WHERE T.LAST_NAME = 'Matos';

16 DISPLAY LAST_NAME AND SALARY FOR EMPLOYEES AND HIS/HER MANAGERIAL COLLEGUE.
SELECT E.FIRST_NAME , E.SALARY , C.FIRST_NAME , C.SALARY
FROM EMPLOYEES E
JOIN EMPLOYEES C ON C.DEPARTMENT_ID = E.DEPARTMENT_ID AND E.EMPLOYEE_ID > C.EMPLOYEE_ID;

17 DISPLAY FIRST_NAME , SALARY AND SALARY DIFFERENCE FOR EMPLOYEE AND HIS/HER JOB COLLEGUE.
SELECT E.FIRST_NAME , E.SALARY , C.FIRST_NAME , C.SALARY , ABS(C.SALARY - E.SALARY) SALARY_DIFF
FROM EMPLOYEES E
JOIN EMPLOYEES C ON C.DEPARTMENT_ID = E.DEPARTMENT_ID AND E.EMPLOYEE_ID > C.EMPLOYEE_ID;

18 DISPLAY FIRST_NAME , HIRE_DATE FOR THE EMPLOYEE WHO WERE HIRED IN THE SAME QUARTER IN WHICH Hunold HIRED. (IGNORE YEAR AND DATE)
SELECT E.FIRST_NAME , E.HIRE_DATE
FROM EMPLOYEES E
JOIN EMPLOYEES T ON TO_CHAR(E.HIRE_DATE,'Q') = TO_CHAR(T.HIRE_DATE,'Q')
WHERE T.LAST_NAME = 'Hunold';

19 DISPLAY FIRST_NAME , SALARY AND SALARY DIFFERENT FOR EMPLOYEE AND HIS/HER JOB COLLEGUE. (ALL DIFFERENCES MUST BE DISPLAYED AS POSITIVE VALUE ONLY.)
SELECT E.FIRST_NAME , E.SALARY , C.FIRST_NAME , C.SALARY , CASE WHEN C.SALARY - E.SALARY < 0 THEN E.SALARY - C.SALARY ELSE C.SALARY - E.SALARY END SALARY_DIFF
FROM EMPLOYEES E
JOIN EMPLOYEES C ON C.DEPARTMENT_ID = E.DEPARTMENT_ID AND E.EMPLOYEE_ID > C.EMPLOYEE_ID;


20. DISPLAY FIRST_NAME , SALARY , GRADE_LEVEL FOR EMPLOYEE AND COLLEGUE. CONSIDER COLLEGUE WHO GETS SALARY IN THE SAME GRADE_LEVEL.
SELECT E.FIRST_NAME , E.SALARY , EJG.GRADE_LEVEL , C.FIRST_NAME , C.SALARY , CJG.GRADE_LEVEL
FROM EMPLOYEES E
JOIN EMPLOYEES C ON C.DEPARTMENT_ID = E.DEPARTMENT_ID AND E.EMPLOYEE_ID > C.EMPLOYEE_ID
JOIN JOB_GRADES EJG ON E.SALARY BETWEEN EJG.LOWEST_SAL AND EJG.HIGHEST_SAL
JOIN JOB_GRADES CJG ON C.SALARY BETWEEN CJG.LOWEST_SAL AND CJG.HIGHEST_SAL
WHERE CJG.GRADE_LEVEL = EJG.GRADE_LEVEL;
