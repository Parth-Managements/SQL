1 DISPLAY FIRST_NAME , DEPARTMENT_ID , SALARY , HIGHEST SALARY PAID TO THE CORRESPONDING DEPARTMENT_ID
SELECT FIRST_NAME , DEPARTMENT_ID  , SALARY , "MAX(SALARY)" FROM EMPLOYEES 
NATURAL JOIN (SELECT DEPARTMENT_ID , MAX(SALARY) FROM EMPLOYEES GROUP BY DEPARTMENT_ID);

2 DISPLAY LAST_NAME , JOB_ID , SALARY , LOWEST SALARY PAID TO THE CORRESPONDING JOB_ID
SELECT LAST_NAME , JOB_ID  , SALARY , "MIN(SALARY)" FROM EMPLOYEES 
NATURAL JOIN (SELECT JOB_ID , MIN(SALARY) FROM EMPLOYEES GROUP BY JOB_ID);

3 DISPLAY FIRST_NAME , MANAGER_ID , SALARY , AVERAGE SALARY PAID TO THE CORRESPONDING MANAGER_ID
SELECT FIRST_NAME , MANAGER_ID  , SALARY , "AVG(SALARY)" FROM EMPLOYEES 
NATURAL JOIN (SELECT MANAGER_ID , AVG(SALARY) FROM EMPLOYEES GROUP BY MANAGER_ID);

4 DISPLAY FIRST_NAME , HIRE YEAR , SALARY , HIRE YEAR WISE LOWEST SALARY , HIRE YEAR WISE HIGHEST SALARY
SELECT E.FIRST_NAME , TO_CHAR(E.HIRE_DATE,'RRRR') HIRE_YEAR , E.SALARY , "MIN(SALARY)" , MAX_SALARY FROM EMPLOYEES E 
JOIN (SELECT TO_CHAR(HIRE_DATE,'RRRR') HIRE_YEAR , MIN(SALARY) , MAX(SALARY) MAX_SALARY FROM EMPLOYEES GROUP BY TO_CHAR(HIRE_DATE,'RRRR'))T ON TO_CHAR(E.HIRE_DATE,'RRRR') = T.HIRE_YEAR;

5 DISPLAY FIRST_NAME , SALARY , DEPARTMENT_ID , DEPARTMENT WISE HIGHEST SALARY , JOB_ID , JOB WISE LOWEST SALARY
SELECT FIRST_NAME , DEPARTMENT_ID  , SALARY , "MIN(SALARY)" , MAX_SALARY FROM EMPLOYEES 
NATURAL JOIN (SELECT DEPARTMENT_ID , MIN(SALARY) , MAX(SALARY) MAX_SALARY FROM EMPLOYEES GROUP BY DEPARTMENT_ID);

6 DISPLAY LAST_NAME , DEPARTMENT_ID , SALARY , LOWEST SALARY , HIGHEST SALARY FOR THE CORRESPONDING DEPARTMENT_ID
SELECT LAST_NAME , DEPARTMENT_ID , SALARY , "MIN(SALARY)" , MAX_SALARY FROM EMPLOYEES
NATURAL JOIN (SELECT DEPARTMENT_ID , MIN(SALARY) , MAX(SALARY) MAX_SALARY FROM EMPLOYEES GROUP BY DEPARTMENT_ID);

7 DISPLAY FIRST_NAME , JOB_ID , SALARY , HIGHEST SALARY , DIFFERENCE (HIGHEST SALARY - SALARY) FOR THE CORRESPONDING JOB_ID
SELECT FIRST_NAME , JOB_ID  , SALARY , "MAX(SALARY)" , MAX_SALARY - SALARY DIFF_SAL FROM EMPLOYEES 
NATURAL JOIN (SELECT JOB_ID , MAX(SALARY) , MAX(SALARY) MAX_SALARY FROM EMPLOYEES GROUP BY JOB_ID);

8 DISPLAY FIRST_NAME , LAST_NAME , DEPARTMENT_ID , JOB_ID , NO. OF EMPLOYEES WORKING FOR DEPARTMENT_ID , NO. OF EMPLOYEES WORKING FOR JOB_ID
SELECT FIRST_NAME , LAST_NAME , DEPARTMENT_ID , JOB_ID , CNT_DP , CNT_JB FROM EMPLOYEES
NATURAL JOIN (SELECT DEPARTMENT_ID , COUNT(*) CNT_DP FROM EMPLOYEES GROUP BY DEPARTMENT_ID)
NATURAL JOIN (SELECT JOB_ID , COUNT(*) CNT_JB FROM EMPLOYEES GROUP BY JOB_ID);

9 DISPLAY FIRST_NAME , JOB_ID , SALARY , TOTAL SALARY PAID TO JOB_ID , %OF SALARY EMPLOYEE GET FOR THE CORRESPONDING JOB_ID
SELECT FIRST_NAME , JOB_ID , SALARY , SUM_SALARY , TRUNC((SALARY/SUM_SALARY) * 100 ) PERCENT_SAL FROM EMPLOYEES
NATURAL JOIN (SELECT JOB_ID , SUM(SALARY) SUM_SALARY FROM EMPLOYEES GROUP BY JOB_ID); 

10 DISPLAY FIRST_NAME , DEPARTMENT_ID , SALARY , LOWEST SALARY , DIFFERENCE IN % FOR THE CORRESPONDING DEPARTMENT_ID
SELECT FIRST_NAME , DEPARTMENT_ID , SALARY , MIN_SALARY , TRUNC((MIN_SALARY/SALARY) * 100) PERCENT_SAL FROM EMPLOYEES
NATURAL JOIN (SELECT DEPARTMENT_ID , MIN(SALARY) MIN_SALARY FROM EMPLOYEES GROUP BY DEPARTMENT_ID); 

11 DISPLAY FIRST_NAME , SALARY , DEPARTMENT_ID , DEPT WISE HIGHEST SALARY , JOB_ID , JOB WISE LOWEST SALARY
SELECT FIRST_NAME , SALARY , DEPARTMENT_ID , DEPT_MAX_SALARY , JOB_ID , JOB_MIN_SALARY FROM EMPLOYEES
NATURAL JOIN (SELECT DEPARTMENT_ID , MAX(SALARY) DEPT_MAX_SALARY FROM EMPLOYEES GROUP BY DEPARTMENT_ID)
NATURAL JOIN (SELECT JOB_ID , MIN(SALARY) JOB_MIN_SALARY FROM EMPLOYEES GROUP BY JOB_ID);

12 DISPLAY LAST_NAME , SALARY , DEPARTMENT_ID , DEPT WISE AVERAGE SALARY , MANAGER_ID , MANAGER WISE NO. OF EMPLOYEES
SELECT LAST_NAME , SALARY , DEPARTMENT_ID , DEPT_AVG_SALARY , MANAGER_ID , MGR_CNT_EMP FROM EMPLOYEES
NATURAL JOIN (SELECT DEPARTMENT_ID , AVG(SALARY) DEPT_AVG_SALARY FROM EMPLOYEES GROUP BY DEPARTMENT_ID)
NATURAL JOIN (SELECT MANAGER_ID , COUNT(*) MGR_CNT_EMP FROM EMPLOYEES GROUP BY MANAGER_ID);

13 DISPLAY LAST_NAME , SALARY , MANAGER_ID , MGR WISE AVERAGE SALARY , JOB_ID , JOB WISE NO. OF EMPLOYEES
SELECT LAST_NAME , SALARY , MANAGER_ID , MGR_AVG_SALARY , JOB_ID , JOB_CNT_EMP FROM EMPLOYEES
NATURAL JOIN (SELECT MANAGER_ID , MAX(SALARY) DEPT_AVG_SALARY FROM EMPLOYEES GROUP BY DEPARTMENT_ID)
NATURAL JOIN (SELECT JOB_ID , COUNT(*) JOB_CNT_EMP FROM EMPLOYEES GROUP BY JOB_ID);

14 DISPLAY LAST_NAME , SALARY , DEPARTMENT_ID , DEPT WISE LOWEST SALARY , JOB_ID , JOB WISE AVERAGE SALARY, MANAGER_ID , MANAGER WISE HIGHEST SALARY
SELECT LAST_NAME , SALARY , DEPARTMENT_ID , DEPT_MIN_SALARY , JOB_ID , JOB_AVG_SALARY , MANAGER_ID , MGR_MAX_SALARY FROM EMPLOYEES
NATURAL JOIN (SELECT DEPARTMENT_ID , MIN(SALARY) DEPT_MIN_SALARY FROM EMPLOYEES GROUP BY DEPARTMENT_ID)
NATURAL JOIN (SELECT JOB_ID , AVG(SALARY) JOB_AVG_SALARY FROM EMPLOYEES GROUP BY JOB_ID)
NATURAL JOIN (SELECT MANAGER_ID , MAX(SALARY) MGR_MAX_SALARY FROM EMPLOYEES GROUP BY MANAGER_ID);

15 DISPLAY LAST_NAME , SALARY , DEPARTMENT_ID , DEPT WISE HIGHEST SALARY , DEPARTMENT_NAME
SELECT LAST_NAME , SALARY , E.DEPARTMENT_ID , DEPT_MAX_SALARY , DEPARTMENT_NAME FROM EMPLOYEES E 
JOIN DEPARTMENTS D ON D.DEPARTMENT_ID = E.DEPARTMENT_ID
JOIN (SELECT DEPARTMENT_ID , MAX(SALARY) DEPT_MAX_SALARY FROM EMPLOYEES GROUP BY DEPARTMENT_ID)T ON T.DEPARTMENT_ID = E.DEPARTMENT_ID;

16 DISPLAY LAST_NAME , SALARY , DEPARTMENT_ID , DEPT WISE AVERAGE SALARY , MANAGER_ID , MANAGER WISE NO. OF EMPLOYEES , DEPARTMENT_NAME , CITY
SELECT LAST_NAME , SALARY , E.DEPARTMENT_ID , DEPT_AVG_SALARY , E.MANAGER_ID , MGR_CNT_EMP , DEPARTMENT_NAME , CITY FROM EMPLOYEES E
JOIN DEPARTMENTS D ON D.DEPARTMENT_ID = E.DEPARTMENT_ID
JOIN LOCATIONS L ON L.LOCATION_ID = D.LOCATION_ID
JOIN (SELECT DEPARTMENT_ID , AVG(SALARY) DEPT_AVG_SALARY FROM EMPLOYEES GROUP BY DEPARTMENT_ID)T ON T.DEPARTMENT_ID = E.DEPARTMENT_ID
JOIN (SELECT MANAGER_ID , COUNT(*) MGR_CNT_EMP FROM EMPLOYEES GROUP BY MANAGER_ID)R ON R.MANAGER_ID = E.MANAGER_ID; 

17 DISPLAY LAST_NAME , SALARY , MANAGER_ID , MGR WISE AVERAGE SALARY , JOB_ID , JOB WISE NO. OF EMPLOYEES , DEPARTMENT_NAME , CITY , COUNTRY_NAME
SELECT LAST_NAME , SALARY , E.MANAGER_ID , MGR_AVG_SALARY , E.JOB_ID , JOB_CNT_EMP , DEPARTMENT_NAME , CITY , COUNTRY_NAME FROM EMPLOYEES E
JOIN DEPARTMENTS D ON D.DEPARTMENT_ID = E.DEPARTMENT_ID
JOIN LOCATIONS L ON L.LOCATION_ID = D.LOCATION_ID
JOIN COUNTRIES C ON C.COUNTRY_ID = L.COUNTRY_ID
JOIN (SELECT MANAGER_ID , AVG(SALARY) MGR_AVG_SALARY FROM EMPLOYEES GROUP BY MANAGER_ID)R ON R.MANAGER_ID = E.MANAGER_ID
JOIN (SELECT JOB_ID , COUNT(*) JOB_CNT_EMP FROM EMPLOYEES GROUP BY JOB_ID)I ON I.JOB_ID = E.JOB_ID;
 
18 DISPLAY LAST_NAME , SALARY , DEPARTMENT_ID , DEPT WISE LOWEST SALARY , JOB_ID , JOB WISE AVERAGE SALARY, MANAGER_ID , MANAGER WISE HIGHEST SALARY , DEPARTMENT_NAME , CITY , COUNTRY_NAME , REGION_NAME
SELECT LAST_NAME , SALARY , E.DEPARTMENT_ID , DEPT_MIN_SALARY , E.JOB_ID , JOB_AVG_SALARY , E.MANAGER_ID , MGR_MAX_SALARY , DEPARTMENT_NAME , CITY , COUNTRY_NAME , REGION_NAME FROM EMPLOYEES E
JOIN DEPARTMENTS D ON D.DEPARTMENT_ID = E.DEPARTMENT_ID
JOIN LOCATIONS L ON L.LOCATION_ID = D.LOCATION_ID
JOIN COUNTRIES C ON C.COUNTRY_ID = L.COUNTRY_ID
JOIN REGIONS R ON R.REGION_ID = C.REGION_ID
JOIN (SELECT DEPARTMENT_ID , MIN(SALARY) DEPT_MIN_SALARY FROM EMPLOYEES GROUP BY DEPARTMENT_ID)T ON T.DEPARTMENT_ID = E.DEPARTMENT_ID
JOIN (SELECT JOB_ID , AVG(SALARY) JOB_AVG_SALARY FROM EMPLOYEES GROUP BY JOB_ID)F ON F.JOB_ID = E.JOB_ID
JOIN (SELECT MANAGER_ID , MAX(SALARY) MGR_MAX_SALARY FROM EMPLOYEES GROUP BY MANAGER_ID)Q ON Q.MANAGER_ID = E.MANAGER_ID;
